async function fetchCSV(url){const r=await fetch(url);if(!r.ok)throw new Error();return r.text()}

function parseCSV(t){

  return t.trim().split('\n').map(row=>{

    const c=[];let cur='',q=false;

    for(const ch of row){if(ch==='"'){q=!q}else if(ch===','&&!q){c.push(cur.trim());cur=''}else cur+=ch}

    c.push(cur.trim());return c;

  });

}

function parsePicks(t){

  const rows=parseCSV(t);if(!rows.length)return;

  const hdr=rows[0];players=[];

  rows.slice(1).forEach(row=>{

    const name=(row[0]||'').trim();if(!name)return;

    const tribe=[],schoolyard=[];

    hdr.forEach((h,i)=>{

      const v=(row[i]||'').trim();if(!v)return;

      if(/^Tribe\s*\d+$/i.test(h))tribe.push(v);

      else if(/^Schoolyard\s*\d+$/i.test(h))schoolyard.push(v);

    });

    players.push({name,tribe,schoolyard});

  });

}

function parsePoints(t){

  const rows=parseCSV(t);if(!rows.length)return;

  const hdr=rows[0].map(h=>h.trim());

  episodeCols=hdr.slice(1).filter(h=>/^Ep\s*#?\d+/i.test(h));

  castawayPoints={};

  rows.slice(1).forEach(row=>{

    const c=(row[0]||'').trim();if(!c)return;

    const eps={};

    hdr.forEach((h,i)=>{if(/^Ep\s*#?\d+/i.test(h)){const v=parseFloat(row[i]);eps[h]=isNaN(v)?0:v}});

    castawayPoints[c.toLowerCase()]=eps;

  });

}

function parseStatus(t){

  statusMap={};

  parseCSV(t).slice(1).forEach(row=>{

    const n=(row[0]||'').trim();if(!n)return;

    statusMap[n.toLowerCase()]={name:n,active:/^active$/i.test((row[1]||'').trim()),jury:/^yes$/i.test((row[2]||'').trim())};

  });

}


const GAVEL=`<span class="jury-gavel">⚖️</span>`;

function allCasts(p){return[...(p.tribe||[]),...(p.schoolyard||[])]}

function playerTotal(p){return allCasts(p).reduce((s,c)=>{const e=castawayPoints[c.toLowerCase()]||{};return s+Object.values(e).reduce((a,b)=>a+b,0)},0)}

function playerEp(p,ep){return allCasts(p).reduce((s,c)=>(s+((castawayPoints[c.toLowerCase()]||{})[ep]||0)),0)}

function getSorted(){return[...players].map(p=>({...p,total:playerTotal(p)})).sort((a,b)=>b.total-a.total)}

function getLastEpDelta(p){

  if(!episodeCols.length)return null;

  return Math.round(playerEp(p,episodeCols[episodeCols.length-1]));

}

function renderHome(){

  const sorted=getSorted();

  const medals=['🥇','🥈','🥉'];

  document.getElementById('top5').innerHTML=sorted.slice(0,5).map((p,i)=>{

    const delta=getLastEpDelta(p);

    let dHtml='';

    if(delta!==null){

      if(delta>0)dHtml=`<span class="t5-delta up">▲ +${delta}</span>`;

      else if(delta<0)dHtml=`<span class="t5-delta down">▼ ${delta}</span>`;

      else dHtml=`<span class="t5-delta same">—</span>`;

    }

    return`<div class="t5">

      <div class="t5-rank">${i<3?medals[i]:i+1}</div>

      <div class="t5-name">${esc(p.name)}</div>

      ${dHtml}

      <div class="t5-pts">${p.total}<em>pts</em></div>

    </div>`;

  }).join('');

}

function chipList(arr,cls){

  if(!arr.length)return`<span class="chip tribe" style="color:var(--muted)">—</span>`;

  return arr.map(c=>{

    const info=statusMap[c.toLowerCase()];

    const out=info&&!info.active;

    const gavel=info?.jury?GAVEL:'';

    return`<span class="chip ${cls}${out?' out':''}">${esc(c)}${gavel}</span>`;

  }).join('');

}

function epPips(p){

  return episodeCols.map(ep=>{

    const v=playerEp(p,ep);

    return`<div class="ep-pip"><span>${ep.replace(/[^0-9]/g,'')}</span><span class="${v===0?'zero':''}">${v}</span></div>`;

  }).join('');

}

}
